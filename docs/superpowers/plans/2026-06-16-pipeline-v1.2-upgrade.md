"""证据层报告类型定义.

V1.2 法律引擎升级引入的证据强度分层报告结构。
"""

from dataclasses import dataclass, field
from typing import Literal

LegalPath = Literal[
    "帮信罪主路径",
    "诈骗罪共同犯罪路径", 
    "掩饰隐瞒犯罪所得路径",
    "规范路径待核实"
]

EvidenceStrength = Literal[
    "直接认知性证据",
    "客观异常事实",
    "认知增强因素",
    "辩解检验材料"
]

@dataclass
class SubjectInfo:
    """单个主体信息."""
    name: str
    role: str
    objective_behavior: str
    cognitive_evidence: list[str] = field(default_factory=list)
    defense: str = ""
    disputes: list[str] = field(default_factory=list)

@dataclass
class EvidenceLayer:
    """证据强度分层.
    
    Attributes:
        strength: 证据强度级别
        facts: 事实列表
        legal_basis: 法律依据
    """
    strength: EvidenceStrength
    facts: list[str]
    legal_basis: str = ""

@dataclass
class BoundaryAlert:
    """边界提醒.
    
    Attributes:
        alert_type: 提醒类型
        description: 描述
        severity: 严重程度 (low/medium/high)
    """
    alert_type: str
    description: str
    severity: Literal["low", "medium", "high"] = "medium"

@dataclass
class EvidenceLayerReport:
    """证据层报告.
    
    V1.2 核心数据结构，包含：
    - 识别的法律路径
    - 多主体信息
    - 证据强度分层
    - 边界提醒
    """
    identified_path: LegalPath
    subjects: list[SubjectInfo] = field(default_factory=list)
    evidence_layers: list[EvidenceLayer] = field(default_factory=list)
    boundary_alerts: list[BoundaryAlert] = field(default_factory=list)
    is_primary_path_bangxin: bool = True
    
    def should_cite_article_287_2(self) -> bool:
        """判断是否应引用《刑法》第287-2条.
        
        仅当识别路径为帮信罪主路径时返回True。
        """
        return self.identified_path == "帮信罪主路径"
    
    def get_scoring_mode(self) -> Literal["definitive", "reference_only"]:
        """获取打分模式.
        
        帮信罪主路径：definitive（决定性）
        其他路径：reference_only（仅参照）
        """
        if self.identified_path == "帮信罪主路径":
            return "definitive"
        return "reference_only"
```

"""规范路径识别模块 (B1).

V1.2 法律引擎升级 - 第一步：识别案件适用的规范路径。
"""

import re
from typing import Literal

from loguru import logger

from app.types.evidence_layer import LegalPath

# 帮信罪关键词
_BANGXIN_KEYWORDS = [
    "明知他人利用信息网络实施犯罪",
    "提供支付结算帮助",
    "提供广告推广",
    "提供通讯传输",
    "提供技术支持",
    "帮助转移资金",
    "代为取款",
    "提供银行卡",
    "出租银行卡",
    "出借银行卡",
]

# 诈骗罪共同犯罪关键词
_FRAUD_JOINT_KEYWORDS = [
    "事先通谋",
    "分工合作",
    "诈骗团伙",
    "共同实施诈骗",
    "参与诈骗",
    "诈骗共犯",
    "电信网络诈骗",
]

# 掩饰隐瞒犯罪所得关键词
_CONCEALMENT_KEYWORDS = [
    "明知是犯罪所得",
    "犯罪所得及其收益",
    "予以窝藏",
    "予以转移",
    "予以收购",
    "代为销售",
    "掩饰隐瞒",
    "洗钱",
]

# 边界提醒关键词（可能超出帮信罪范围）
_BOUNDARY_INDICATORS = [
    "明确知道系诈骗钱款",
    "长期取现分工",
    "上线安排",
    "每日验卡",
    "防止冻结",
    "分开装袋",
    "事先知道诈骗",
    "参与诈骗分工",
]


def _count_keyword_matches(text: str, keywords: list[str]) -> int:
    """统计关键词匹配数量."""
    count = 0
    for kw in keywords:
        if kw in text:
            count += 1
    return count


def _has_boundary_indicators(text: str) -> bool:
    """检查是否包含边界提醒指标."""
    for indicator in _BOUNDARY_INDICATORS:
        if indicator in text:
            return True
    return False


def identify_legal_path(case_text: str) -> LegalPath:
    """识别案件适用的规范路径.
    
    优先级：
    1. 若包含边界提醒指标 → 检查是否更符合诈骗共犯或掩饰隐瞒
    2. 诈骗罪共同犯罪路径（事先通谋、分工合作）
    3. 掩饰隐瞒犯罪所得路径（明知是犯罪所得）
    4. 帮信罪主路径（明知他人利用信息网络实施犯罪）
    5. 规范路径待核实（信息不足）
    
    Args:
        case_text: 案件事实文本
        
    Returns:
        LegalPath: 识别的法律路径
    """
    logger.info("开始规范路径识别 (B1)")
    
    has_boundary = _has_boundary_indicators(case_text)
    
    # 统计各路径关键词匹配
    fraud_score = _count_keyword_matches(case_text, _FRAUD_JOINT_KEYWORDS)
    concealment_score = _count_keyword_matches(case_text, _CONCEALMENT_KEYWORDS)
    bangxin_score = _count_keyword_matches(case_text, _BANGXIN_KEYWORDS)
    
    logger.debug(
        f"路径识别得分: fraud={fraud_score}, concealment={concealment_score}, "
        f"bangxin={bangxin_score}, has_boundary={has_boundary}"
    )
    
    # 优先级1: 诈骗罪共同犯罪（事先通谋、分工合作等强指标）
    if fraud_score >= 2 or (fraud_score >= 1 and has_boundary):
        logger.info("识别路径: 诈骗罪共同犯罪路径")
        return "诈骗罪共同犯罪路径"
    
    # 优先级2: 掩饰隐瞒犯罪所得
    if concealment_score >= 2 or (concealment_score >= 1 and bangxin_score == 0):
        logger.info("识别路径: 掩饰隐瞒犯罪所得路径")
        return "掩饰隐瞒犯罪所得路径"
    
    # 优先级3: 帮信罪主路径
    if bangxin_score >= 1 and fraud_score == 0:
        logger.info("识别路径: 帮信罪主路径")
        return "帮信罪主路径"
    
    # 优先级4: 规范路径待核实
    logger.info("识别路径: 规范路径待核实")
    return "规范路径待核实"
```

"""多主体分层模块 (B2).

V1.2 法律引擎升级 - 第二步：识别并分层多个被告人。
"""

import re
from typing import List

from loguru import logger

from app.types.evidence_layer import SubjectInfo

# 被告人姓名模式（张某、李某、王某某等）
_SUBJECT_NAME_PATTERN = re.compile(r"被告人([赵钱孙李周吴郑王张刘陈杨黄赵周吴徐孙马朱胡郭林何高罗郑梁谢宋唐许韩冯邓曹彭曾田萧潘袁蔡蒋余于杜叶程苏魏吕丁任沈姚卢姜崔钟谭陆汪范金石廖贾夏韦付方白邹孟熊秦邱江尹薛闫段雷侯龙史贺顾毛郝孔邵毛常万顾赖武康])某{1,2}")

# 客观行为关键词
_BEHAVIOR_KEYWORDS = [
    "提供银行卡", "出租银行卡", "出借银行卡",
    "代为取款", "负责取款", "取现",
    "转账", "转移资金", "支付结算",
    "联系上线", "联系下线", "介绍",
    "提供技术支持", "提供账户",
    "帮助转移", "代为销售",
]

# 认知证据关键词
_COGNITION_KEYWORDS = [
    "明知", "知道", "应当知道",
    "被告知", "认识到",
    "聊天记录显示", "通话记录显示",
    "供述", "承认",
]

# 辩解关键词
_DEFENSE_KEYWORDS = [
    "辩称", "辩解", "不认可",
    "不知情", "不清楚", "不知道",
    "被蒙蔽", "被骗",
]


def _extract_subject_names(text: str) -> list[str]:
    """提取被告人姓名."""
    matches = _SUBJECT_NAME_PATTERN.findall(text)
    # 去重并保持顺序
    seen = set()
    names = []
    for name in matches:
        if name not in seen:
            seen.add(name)
            names.append(name)
    return names


def _extract_behavior(text: str, subject_name: str) -> str:
    """提取指定主体的客观行为."""
    # 简单策略：查找姓名后的行为描述
    # 实际应使用更复杂的NLP或规则
    behavior_parts = []
    for kw in _BEHAVIOR_KEYWORDS:
        if kw in text:
            # 检查是否在主体相关上下文中
            pattern = f"{subject_name}.*?{kw}|{kw}.*?{subject_name}"
            if re.search(pattern, text, re.DOTALL):
                behavior_parts.append(kw)
    return "、".join(behavior_parts) if behavior_parts else "行为待确认"


def _extract_cognition_evidence(text: str, subject_name: str) -> list[str]:
    """提取指定主体的认知证据."""
    evidence = []
    for kw in _COGNITION_KEYWORDS:
        if kw in text:
            pattern = f"{subject_name}.*?{kw}|{kw}.*?{subject_name}"
            if re.search(pattern, text, re.DOTALL):
                evidence.append(kw)
    return evidence


def _extract_defense(text: str, subject_name: str) -> str:
    """提取指定主体的辩解."""
    for kw in _DEFENSE_KEYWORDS:
        if kw in text:
            pattern = f"{subject_name}.*?{kw}|{kw}.*?{subject_name}"
            if re.search(pattern, text, re.DOTALL):
                return kw
    return ""


def stratify_subjects(case_text: str) -> list[SubjectInfo]:
    """对案件中的多个主体进行分层分析.
    
    识别所有被告人，并为每个被告人提取：
    - 角色定位
    - 客观行为
    - 认知证据
    - 辩解或争议
    
    Args:
        case_text: 案件事实文本
        
    Returns:
        list[SubjectInfo]: 主体信息列表
    """
    logger.info("开始多主体分层 (B2)")
    
    names = _extract_subject_names(case_text)
    logger.info(f"识别到 {len(names)} 个被告人: {names}")
    
    subjects = []
    for name in names:
        behavior = _extract_behavior(case_text, name)
        cognition = _extract_cognition_evidence(case_text, name)
        defense = _extract_defense(case_text, name)
        
        subject = SubjectInfo(
            name=name,
            role="被告人",
            objective_behavior=behavior,
            cognitive_evidence=cognition,
            defense=defense,
        )
        subjects.append(subject)
        
        logger.debug(
            f"主体 {name}: behavior={behavior}, "
            f"cognition={cognition}, defense={defense}"
        )
    
    return subjects
```

from typing import List

from loguru import logger

from app.types.evidence_layer import EvidenceLayer, EvidenceLayerReport, EvidenceStrength

_DIRECT_COGNITION_KEYWORDS = [

    "同案犯供述", "共犯供述",
    "证人证言证实", "被害人陈述",

def _build_layer(
    strength: EvidenceStrength,
    keywords: list[str],
    case_text: str,
    legal_basis: str = "",
) -> EvidenceLayer | None:
    """构建单个证据层."""
    facts = []
    for kw in keywords:
        if kw in case_text:
            facts.append(kw)
    
    if not facts:
        return None
    
    return EvidenceLayer(
        strength=strength,
        facts=facts,
        legal_basis=legal_basis,
    )


def build_evidence_layers(case_text: str) -> list[EvidenceLayer]:
    """构建证据强度分层.
    
    将证据分为四层：
    1. 直接认知性证据（被告人供述、同案犯供述等）
    2. 客观异常事实（账户异常、交易异常等）
    3. 认知增强因素（被告知、被提醒等）
    4. 辩解检验材料（辩解及其合理性检验）
    
    Args:
        case_text: 案件事实文本
        
    Returns:
        list[EvidenceLayer]: 证据层列表
    """
    logger.info("开始证据强度分层 (B3)")
    
    layers = []
    
    # 第一层：直接认知性证据
    layer1 = _build_layer(
        "直接认知性证据",
        _DIRECT_COGNITION_KEYWORDS,
        case_text,
        legal_basis="《刑法》第287-2条：明知他人利用信息网络实施犯罪",
    )
    if layer1:
        layers.append(layer1)
        logger.debug(f"直接认知性证据: {layer1.facts}")
    
    # 第二层：客观异常事实
    layer2 = _build_layer(
        "客观异常事实",
        _OBJECTIVE_ANOMALY_KEYWORDS,
        case_text,
        legal_basis="客观行为异常度分析",
    )
    if layer2:
        layers.append(layer2)
        logger.debug(f"客观异常事实: {layer2.facts}")
    
    # 第三层：认知增强因素
    layer3 = _build_layer(
        "认知增强因素",
        _COGNITIVE_ENHANCEMENT_KEYWORDS,
        case_text,
        legal_basis="认知匹配度增强因素",
    )
    if layer3:
        layers.append(layer3)
        logger.debug(f"认知增强因素: {layer3.facts}")
    
    # 第四层：辩解检验材料
    layer4 = _build_layer(
        "辩解检验材料",
        _DEFENSE_TEST_KEYWORDS,
        case_text,
        legal_basis="辩解合理性检验",
    )
    if layer4:
        layers.append(layer4)
        logger.debug(f"辩解检验材料: {layer4.facts}")
    
    logger.info(f"构建完成 {len(layers)} 个证据层")
    return layers


def guard_against_single_layer_override(report: EvidenceLayerReport) -> bool:
    """防止单层覆盖机制.
    
    规则：仅当存在直接认知性证据时，才能认定主观明知。
    仅有客观异常事实不足以单独认定明知。
    
    Args:
        report: 证据层报告
        
    Returns:
        bool: True表示可以认定明知，False表示不可以
    """
    has_direct_cognition = any(
        layer.strength == "直接认知性证据" 
        for layer in report.evidence_layers
    )
    
    if not has_direct_cognition:
        logger.warning(
            "防止单层覆盖: 缺少直接认知性证据，"
            "仅凭客观异常事实不足以认定主观明知"
        )
        return False
    
    return True
```

```python
async def _three_dimension_scoring(
    evidence_report: EvidenceLayerReport,
    case_text: str,
) -> tuple[dict[str, Any], dict[str, Any], dict[str, Any]]:
    """基于EvidenceLayerReport的三维度打分.
    
    V1.2 升级：
    - 输入参数由原始标签数据改为EvidenceLayerReport
    - 集成B3模块的guard_against_single_layer_override机制
    - 非帮信罪路径下仅输出参照信息，不引用《刑法》第287-2条
    
    Args:
        evidence_report: 证据层报告
        case_text: 案件文本
        
    Returns:
        tuple: (dim1_result, dim2_result, dim3_result)
    """
    logger.info("开始三维度打分（基于EvidenceLayerReport）")
    
    # 检查是否可以使用单层覆盖
    can_override = guard_against_single_layer_override(evidence_report)
    
    # 维度1：客观行为异常度
    dim1_reasoning = "客观行为异常度分析：\n"
    for layer in evidence_report.evidence_layers:
        if layer.strength == "客观异常事实":
            dim1_reasoning += f"- 客观异常事实: {', '.join(layer.facts)}\n"
    
    dim1_tier = "T2"  # 默认
    if any(l.strength == "客观异常事实" for l in evidence_report.evidence_layers):
        dim1_tier = "T3"
    
    dim1_result = {
        "tier": dim1_tier,
        "reasoning": dim1_reasoning,
        "key_indicators": ["客观行为异常"],
        "triggered_rules": [],
    }
    
    # 维度2：认知匹配度
    dim2_reasoning = "认知匹配度分析：\n"
    has_direct_cognition = False
    for layer in evidence_report.evidence_layers:
        if layer.strength == "直接认知性证据":
            has_direct_cognition = True
            dim2_reasoning += f"- 直接认知证据: {', '.join(layer.facts)}\n"
        elif layer.strength == "认知增强因素":
            dim2_reasoning += f"- 认知增强: {', '.join(layer.facts)}\n"
    
    dim2_tier = "T2"
    if has_direct_cognition and can_override:
        dim2_tier = "T3"
    
    # 根据路径决定是否引用法条
    if evidence_report.should_cite_article_287_2():
        dim2_reasoning += "\n法律依据: 《刑法》第287-2条"
    else:
        dim2_reasoning += "\n注: 非帮信罪路径，三维度仅作主观明知证据结构参照"
    
    dim2_result = {
        "tier": dim2_tier,
        "reasoning": dim2_reasoning,
        "pattern_match": evidence_report.identified_path,
        "triggered_rules": [],
    }
    
    # 维度3：辩解合理性检验
    dim3_reasoning = "辩解合理性检验：\n"
    for layer in evidence_report.evidence_layers:
        if layer.strength == "辩解检验材料":
            dim3_reasoning += f"- 辩解材料: {', '.join(layer.facts)}\n"
    
    if evidence_report.boundary_alerts:
        dim3_reasoning += "\n边界提醒:\n"
        for alert in evidence_report.boundary_alerts:
            dim3_reasoning += f"- [{alert.severity}] {alert.description}\n"
    
    dim3_tier = "T2"
    dim3_result = {
        "tier": dim3_tier,
        "reasoning": dim3_reasoning,
        "contradictions": [],
        "triggered_rules": [],
    }
    
    logger.info(
        f"三维度打分完成: dim1={dim1_tier}, dim2={dim2_tier}, dim3={dim3_tier}, "
        f"path={evidence_report.identified_path}, "
        f"scoring_mode={evidence_report.get_scoring_mode()}"
    )
    
    return dim1_result, dim2_result, dim3_result
```
